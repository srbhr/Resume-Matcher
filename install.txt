# Resume Matcher - Development Install Guide
# ==========================================
# This file lists only the main packages (no versions).
# Transitive dependencies are handled automatically by package managers.

# ============================================
# BACKEND (Python) - apps/backend/
# ============================================
# Install with: uv pip install <packages>
# Or copy-paste the one-liner below

# One-liner install:
# uv pip install fastapi uvicorn python-multipart pydantic pydantic-settings tinydb litellm markitdown python-dotenv

# Individual packages:
# ---------------------
# fastapi           - Web framework
# uvicorn           - ASGI server
# python-multipart  - File upload handling
# pydantic          - Data validation
# pydantic-settings - Environment config
# tinydb            - JSON database
# litellm           - Multi-provider LLM
# markitdown        - PDF/DOCX to Markdown
# python-dotenv     - .env file loading

# Dev dependencies (optional):
# uv pip install pytest httpx


# ============================================
# FRONTEND (Node.js) - apps/frontend/
# ============================================
# Install with: npm install <packages>
# Or copy-paste the one-liner below

# One-liner install (dependencies):
# npm install next react react-dom lucide-react tw-animate-css

# One-liner install (devDependencies):
# npm install -D tailwindcss typescript eslint prettier eslint-config-next eslint-config-prettier eslint-plugin-prettier @tailwindcss/postcss @types/node @types/react @types/react-dom @eslint/eslintrc

# Individual packages:
# ---------------------
# Dependencies:
# next              - React framework
# react             - UI library
# react-dom         - React DOM renderer
# lucide-react      - Icon library
# tw-animate-css    - Tailwind animations

# Dev Dependencies:
# tailwindcss       - Utility CSS framework
# typescript        - Type checking
# eslint            - Linting
# prettier          - Code formatting
# eslint-config-next      - Next.js ESLint config
# eslint-config-prettier  - Prettier ESLint integration
# eslint-plugin-prettier  - Prettier as ESLint rule
# @tailwindcss/postcss    - Tailwind PostCSS plugin
# @types/node       - Node.js types
# @types/react      - React types
# @types/react-dom  - React DOM types
# @eslint/eslintrc  - ESLint config utilities


# ============================================
# QUICK START
# ============================================
# Backend:
#   cd apps/backend
#   uv venv
#   source .venv/bin/activate  # or .venv\Scripts\activate on Windows
#   uv pip install fastapi uvicorn python-multipart pydantic pydantic-settings tinydb litellm markitdown python-dotenv
#   cp .env.example .env
#   uvicorn app.main:app --reload

# Frontend:
#   cd apps/frontend
#   npm install
#   npm run dev
